Esempio di metodo GetRows

Nell'esempio riportato di seguito il metodo GetRows viene utilizzato per restituire una matrice a due dimensioni contenente tutte le righe di dati di un oggetto Recordset:

Sub MatriceRighe()
    Dim rst As New ADODB.Recordset, strSQL As String
    Dim varRecords As Variant, intI As Integer, intJ As Integer
            
    ' Costruisce istruzione SQL che restituisce campi specificati.
    strSQL = "SELECT Nome, Cognome, DataAssunzione " _
        & "FROM Dipendenti"
    ' Apre oggetto Recordset di tipo dynaset.
    rst.Open strSQL, CurrentProject.Connection, adOpenKeyset, adLockOptimistic
    ' Passa alla fine del recordset.
    rst.MoveLast
    ' Ritorna al primo record.
    rst.MoveFirst
    ' Restituisce tutte le righe in una matrice.
    varRecords = rst.GetRows(rst.RecordCount)
    ' Trova limite superiore di seconda dimensione.
    For intI = 0 To UBound(varRecords, 2)
        Debug.Print
        ' Trova limite superiore di prima dimensione.
        For intJ = 0 To UBound(varRecords, 1)
            ' Stampa i dati di ciascuna riga della matrice.
            Debug.Print varRecords(intJ, intI)
        Next intJ
    Next intI
    rst.Close
    
End Sub
