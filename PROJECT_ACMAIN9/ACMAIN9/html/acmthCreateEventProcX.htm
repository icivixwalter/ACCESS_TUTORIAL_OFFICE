Esempio di metodi CreateEventProc, InsertLines

Nell'esempio riportato di seguito viene indicato come creare una nuova maschera, viene aggiunto un pulsante di comando e creata una routine evento Clic per il pulsante di comando:

Function ClickEventProc() As Boolean
    Dim frm As Form, ctl As Control, mdl As Module
    Dim lngReturn As Long

    On Error GoTo Error_ClickEventProc
    ' Crea la nuova maschera.
    Set frm = CreateForm
    ' Crea il pulsante di comando sulla maschera.
    Set ctl = CreateControl(frm.Name, acCommandButton, , , , _
         1000, 1000)
    ctl.Caption = "Fare clic qui"
    ' Restituisce il riferimento al modulo della maschera.
    Set mdl = frm.Module
    ' Aggiunge una routine evento.
    lngReturn = mdl.CreateEventProc("Clic", ctl.Name)
    ' Inserisce il testo nel corpo della routine.
    mdl.InsertLines lngReturn + 1, vbTab & "MsgBox ""In arrivo!"""
    ClickEventProc = True

Exit_ClickEventProc:
    Exit Function

Error_ClickEventProc:
    MsgBox Err & " :" & Err.Description
    ClickEventProc = False
    Resume Exit_ClickEventProc
End Function
