Esempio di proprietà PrtMip

Nell'esempio riportato di seguito della proprietà PrtMip viene dimostrato come impostare il report con due colonne orizzontali.

Type str_PRTMIP
    strRGB As String * 28
End Type
Type type_PRTMIP
    intMargineSin As Long
    intMargineSup As Long
    intMargineDes As Long
    intMargineInf As Long
    intSoloDati As Long
    intLarghezza As Long
    intAltezza As Long
    intDimensPredef As Long
    intColonne As Long
    intSpaziaturaCol As Long
    intSpaziaturaRig As Long
    intLayoutElem As Long
    intStampaVeloce As Long
    intFoglioDati As Long
End Type

Sub ColonnePrtMip(strNome As String)
    Dim StringaPrtMip As str_PRTMIP
    Dim PM As type_PRTMIP
    Dim rpt As Report
    Const PM_HORIZONTALCOLS = 1953
    Const PM_VERTICALCOLS = 1954
    DoCmd.OpenReport strNome, acDesign
    Set rpt = Reports(strNome)
    StringaPrtMip.strRGB = rpt.PrtMip
    LSet PM = StringaPrtMip
    PM.intColonne = 2                    ' Crea due colonne.
    PM.intSpaziaturaRig = 0.25 * 1440    ' Imposta 0,63 cm tra righe.
    PM.intSpaziaturaCol = 0.5 * 1440    ' Imposta 1,2 cm tra colonne.
    PM.intLayoutElem = PM_HORIZONTALCOLS
    
    LSet StringaPrtMip = PM            ' Aggiorna la proprietà.
    rpt.PrtMip = StringaPrtMip.strRGB
End Sub

Nell'esempio successivo della proprietà PrtMip viene indicato come impostare tutti i margini a 2,54 cm.

Sub ImpostaMarginiPredefin(strNome As String)
    Dim StringaPrtMip As str_PRTMIP
    Dim PM As type_PRTMIP
    Dim rpt As Report
    DoCmd.OpenReport strNome, acDesign
    Set rpt = Reports(strNome)
    StringaPrtMip.strRGB = rpt.PrtMip
    LSet PM = StringaPrtMip
    PM.intMargineSin = 1 * 1440        ' Imposta margini.
    PM.intMargineSup = 1 * 1440
    PM.intMargineDes = 1 * 1440
    PM.intMarginInf = 1 * 1440
    LSet StringaPrtMip = PM            ' Aggiorna la proprietà.
    rpt.PrtMip = StringaPrtMip.strRGB
End Sub
