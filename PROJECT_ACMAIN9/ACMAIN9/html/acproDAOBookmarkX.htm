Esempio di proprietà Bookmark e Bookmarkable (MDB)

Nell'esempio riportato di seguito si scorrono i record della tabella Impiegati dall'inizio fino alla fine del file e si memorizza il valore della proprietà Bookmark di ogni record in una matrice.

' Include questa istruzione nella sezione Dichiarazioni del modulo.
Option Compare Binary

Sub PosizioniRecord()
    Dim dbs As Database, rst As Recordset, fld As Field
    Dim intI As Integer
    ' Dichiara la matrice per contenere i segnalibri.
    Dim varRecord() As Variant

    ' Restituisce il riferimento al database corrente.
    Set dbs = CurrentDb
    ' Apre oggetto Recordset di tipo tabella.
    Set rst = dbs.OpenRecordset("Impiegati")
    ' Restituisce il riferimento al campo Cognome.
    Set fld = rst.Fields!Cognome
    ' Ridimensiona la matrice in base al valore della proprietà
    ' RecordCount come limite superiore.
    ReDim varRecord(0 To rst.RecordCount - 1)
    intI = 0
    ' Verifica la proprietà Bookmarkable dell'oggetto Recordset.
    If rst.Bookmarkable Then
        Do Until rst.EOF
            ' Completa la matrice con i segnalibri.
            varRecord(intI) = rst.Bookmark
            ' Incrementa il contatore.
            intI = intI + 1
            rst.MoveNext
        Loop
    End If
    rst.Close
    Set dbs = Nothing
End Sub
