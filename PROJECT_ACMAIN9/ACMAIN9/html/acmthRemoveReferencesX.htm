Esempio di metodo Remove (insieme References)

La prima delle due funzioni riportate di seguito aggiunge un riferimento al Controllo calendario nell'insieme References. La seconda funzione rimuove il riferimento al Controllo calendario.

Function AggiungiRiferimento() As Boolean
    Dim ref As Reference, strFile As String

    On Error GoTo Error_AggiungiRiferimento
    strFile = "C:\Windows\System\Mscal.ocx"
    ' Crea un riferimento al Controllo calendario.
    Set ref = References.AddFromFile(strFile)
    AggiungiRiferimento = True

Exit_AggiungiRiferimento:
    Exit Function

Error_AggiungiRiferimento:
    MsgBox Err & ": " & Err.Description
    AggiungiRiferimento = False
    Resume Exit_AggiungiRiferimento
End Function

Function RimuoviRiferimento() As Boolean
    Dim ref As Reference

    On Error GoTo Error_RimuoviRiferimento
    Set ref = References!MSACAL
    ' Rimuove il riferimento del Controllo calendario.
    References.Remove ref
    RimuoviRiferimento = True

Exit_RimuoviRiferimento:
    Exit Function

Error_RimuoviRiferimento:
    MsgBox Err & ": " & Err.Description
    RimuoviRiferimento = False
    Resume Exit_RimuoviRiferimento
End Function
