Esempio di proprietà Application e DBEngine

Nell'esempio riportato di seguito vengono visualizzate le proprietà DBEngine in una casella di messaggio.

Private Sub Comando1_Click()
  MostraInfoApplicazione Me
End Sub

Function MostraInfoApplicazione(obj As Object) As Integer
  Dim objApp As Object, intI As Integer, strProp As String
  On Error Resume Next
    ' Proprietà Application della maschera.
    Set objApp = obj.Application
    MsgBox "Proprietà Application visibile = " & objApp.Visible
    If objApp.UserControl = True Then
    For intI = 0 To objApp.DBEngine.Properties.Count - 1
      strProp = strProp & objApp.DBEngine.Properties(intI).Name & ", "
    Next intI
    End If
    MsgBox Left(strProp, Len(strProp) - 2) & ".", vbOK, "Proprietà DBEngine"
End Function
