Esempio di metodo RefreshTitleBar

Nell'esempio riportato di seguito viene impostata la proprietà AppTitle del database corrente e applicato il metodo RefreshTitleBar per aggiornare la barra del titolo.

Sub ModificaTitolo()
    Dim obj As Object
    Const conPropNotFoundError = 3270
    
    On Error GoTo ErrorHandler
    ' Restituisce la variabile di oggetto Database che punta al
    ' database corrente.
    Set dbs = CurrentDb
    ' Modifica la barra del titolo.
    dbs.Properties!AppTitle = "Database contatti"
    ' Aggiorna la barra del titolo su schermo.
    Application.RefreshTitleBar
    Exit Sub
    
ErrorHandler:
    If Err.Number = conPropNotFoundError Then
        Set obj = dbs.CreateProperty("TitoloApp", dbText, "Database contatti")
        dbs.Properties.Append obj
    Else
        MsgBox "Errore: " & Err.Number & vbCrLf & Err.Description
    End If
    Resume Next
End Sub
