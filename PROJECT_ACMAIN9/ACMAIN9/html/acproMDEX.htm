Esempio di proprietà MDE

Nell'esempio riportato di seguito viene utilizzata la funzione MDE per controllare la proprietà MDE per il database corrente.

Dim dbs As Object
Set dbs = CurrentDb
If IsItMDE(dbs) <> True Then
  ' Elabora il codice del database del cliente.
End If

Function IsItMDE(dbs as Object) As Boolean
  Dim strMDE As String
  On Error Resume Next
  strMDE = dbs.Properties("MDE")
  If Err = 0 AND strMDE = "T" Then
    ' Questo è un database MDE.
    IsItMDE = True
  Else
    IsITMDE = False
  End if
End Function
