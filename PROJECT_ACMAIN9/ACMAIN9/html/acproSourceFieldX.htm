Esempio di proprietà SourceField e SourceTable (MDB)

Nell'esempio riportato di seguito si crea un oggetto Recordset utilizzando un'istruzione SQL che genera gli alias per i campi in due tabelle diverse del database. In questo esempio si stampa quindi il nome del campo, la tabella e il campo originali.

Sub InfoOrigine()
    Dim dbs As Database, rst As Recordset, fld As Field
    Dim strSQL As String

    ' Restituisce il riferimento al database corrente.
    Set dbs = CurrentDb
    ' Crea istruzione SQL.
    strSQL = "SELECT IDProdotto As CodiceProdotto, " _
        & "NomeCategoria As TipoDiProdotto FROM Categorie " _
        & "INNER JOIN Prodotti ON Categorie.IDCategoria " _
        & " = Prodotti.IDCategoria;"
    ' Apre un oggetto Recordset di tipo dynaset.
    Set rst = dbs.OpenRecordset(strSQL)
    For Each fld In rst.Fields
        ' Stampa il nome del file.
        Debug.Print "Nome del campo: "; fld.Name
        ' Stampa il nome della tabella di origine.
        Debug.Print "Tabella di origine: "; fld.SourceTable
        ' Stampa il nome del campo di origine.
        Debug.Print "Campo di origine: "; fld.SourceField
        Debug.Print
    Next fld
    rst.Close
    Set dbs = Nothing
End Sub
