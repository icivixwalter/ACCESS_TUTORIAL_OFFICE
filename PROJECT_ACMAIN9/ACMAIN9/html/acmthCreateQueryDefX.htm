Esempio di metodo CreateQueryDef (MDB)

Questo esempio utilizza il metodo CreateQueryDef per creare ed eseguire una QueryDef permanente e una temporale. La funzione GetrstTemp è richiesta per l'esecuzione di questa procedura.

Sub CreateQueryDefX()

    Dim dbsNorthwind As Database
    Dim qdfTemp As QueryDef
    Dim qdfNew As QueryDef

    Set dbsNorthwind = OpenDatabase("Northwind.mdb")

    With dbsNorthwind
        ' Crea oggetto QueryDef temporaneo.
        Set qdfTemp = .CreateQueryDef("", _
            "SELECT * FROM Dipendenti")
        ' Apre Recordset e stampa report.
        GetrstTemp qdfTemp
        ' Crea oggetto QueryDef permanente.
        Set qdfNew = .CreateQueryDef("NuovoQueryDef", _
            "SELECT * FROM Categorie")

' Apre Recordset e stampa report.
        GetrstTemp qdfNew
        ' Elimina il nuovo QueryDef perché è soltanto una dimostrazione.
        .QueryDefs.Delete qdfNew.Name
        .Close
    End With

End Sub

Function GetrstTemp(qdfTemp As QueryDef)

    Dim rstTemp As Recordset

    With qdfTemp
        Debug.Print .Name
        Debug.Print "    " & .SQL
        ' Apre Recordset da QueryDef.
        Set rstTemp = .OpenRecordset(dbOpenSnapshot)

        With rstTemp
            ' Riempie i Recordset e stampa il numero di record.

.MoveLast
            Debug.Print "    Numero di record = " & _
                .RecordCount
            Debug.Print
            .Close
        End With

    End With

End Function
