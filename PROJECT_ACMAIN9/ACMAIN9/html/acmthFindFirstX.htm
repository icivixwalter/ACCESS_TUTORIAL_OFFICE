Esempio di metodi FindFirst, FindLast, FindNext e FindPrevious (MDB)

Nell'esempio riportato di seguito viene creato un oggetto Recordset di tipo dynaset e utilizzato il metodo FindFirst per individuare il primo record che soddisfa i criteri specificati. La routine trova quindi gli altri record che soddisfano i criteri.

Sub TrovaRecord()
    Dim dbs As Database, rst As Recordset
    Dim strCriteria As String

    ' Restituisce il riferimento al database corrente.
    Set dbs = CurrentDb
    ' Definisce i criteri di ricerca.
    strCriteria = "[PaeseDestinatario] = 'IT' And " _
        & "[DataOrdine] >= #1-1-95#"
    ' Crea un oggetto Recordset di tipo dynaset basato sulla tabella Ordini.
    Set rst = dbs.OpenRecordset("Ordini", dbOpenDynaset)
    ' Rileva il primo record corrispondente.
    rst.FindFirst strCriteria
    ' Verifica se viene trovato il record.
    If rst.NoMatch Then
        MsgBox "Nessun record trovato."
    Else
        ' Rileva gli altri record corrispondenti.
        Do Until rst.NoMatch
            Debug.Print rst!PaeseDestinatario; "   "; rst!DataOrdine
            rst.FindNext strCriteria
        Loop
    End If
    rst.Close
    Set dbs = Nothing
End Sub
