Esempio di oggetto Recordset e insieme Recordsets (MDB)

Nell'esempio riportato di seguito viene aperto un oggetto Recordset di tipo tabella, un oggetto Recordset di tipo dynaset e un oggetto Recordset di tipo snapshot. Viene quindi visualizzato il valore della proprietà Updatable per gli oggetti Recordset.

Sub NuoviRecordset()
    Dim dbs As Database, rst As Recordset
    Dim rstEmployees As Recordset, rstOrders As Recordset
    Dim rstProducts As Recordset, strSQL As String

    ' Restituisce riferimento a database corrente.
    Set dbs = CurrentDb
    ' Crea oggetto Recordset di tipo tabella.
    Set rstEmployees = dbs.OpenRecordset("Impiegati", dbOpenTable)
    ' Crea oggetto Recordset di tipo dynaset.
    Set rstOrders = dbs.OpenRecordset("Impiegati", dbOpenDynaset)
    ' Crea oggetto Recordset di tipo snapshot.
    Set rstProducts = dbs.OpenRecordset("Prodotti", dbOpenSnapshot)
    ' Stampa valore della proprietà Updatable di ciascun oggetto Recordset.
    For Each rst In dbs.Recordsets
        Debug.Print rst.Name; "   "; rst.Updatable
    Next rst
    ' Rilascia tutte le variabili di oggetto.
    rstEmployees.Close
    rstOrders.Close
    rstProducts.Close
    Set dbs = Nothing
End Sub
