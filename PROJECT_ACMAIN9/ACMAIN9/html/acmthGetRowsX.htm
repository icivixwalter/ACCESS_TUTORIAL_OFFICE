Esempio di metodo GetRows (MDB)

Nell'esempio riportato di seguito viene utilizzato il metodo GetRows per restituire una matrice bidimensionale contenente tutte le righe di dati di un oggetto Recordset:

Sub MatriceRighe()
    Dim dbs As Database, rst As Recordset, strSQL As String
    Dim varRecord As Variant, intI As Integer, intJ As Integer
    
    ' Restituisce il riferimento al database corrente.
    Set dbs = CurrentDb
    ' Costruisce istruzione SQL che restituisce i campi specificati.
    strSQL = "SELECT Nome, Cognome, DataAssunzione " _
        & "FROM Dipendenti"
    ' Apre oggetto Recordset di tipo dynaset.
    Set rst = dbs.OpenRecordset(strSQL)
    ' Passa alla fine del recordset.
    rst.MoveLast
    ' Ritorna al primo record.
    rst.MoveFirst
    ' Restituisce tutte le righe nella matrice.
    varRecord = rst.GetRows(rst.RecordCount)
    ' Trova il limite superiore di seconda dimensione.
    For intI = 0 To UBound(varRecord, 2)
        Debug.Print
        ' Trova il limite superiore di prima dimensione.
        For intJ = 0 To UBound(varRecord, 1)
            ' Stampa i dati di ciascuna riga della matrice.
            Debug.Print varRecord(intJ, intI)
        Next intJ
    Next intI
    rst.Close
    Set dbs = Nothing
End Sub
