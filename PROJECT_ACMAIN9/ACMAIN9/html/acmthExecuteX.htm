Esempio di metodo Execute (MDB)

Nell'esempio riportato di seguito viene eseguita una query di comando e stampato il relativo numero di record:

Sub RecordAggiornati()
    Dim dbs As Database, qdf As QueryDef
    Dim strSQL As String
    
    ' Restituisce il riferimento al database corrente.
    Set dbs = CurrentDb
    strSQL = "UPDATE Dipendenti SET Posizione = " _
        & "'Rappresentante senior' " _
        & "WHERE Posizione = 'Rappresentante';"
    ' Crea il nuovo oggetto QueryDef.
    Set qdf = dbs.CreateQueryDef("AggiornaPosizioni", strSQL)
    ' Esegue QueryDef.
    qdf.Execute
    Debug.Print qdf.RecordsAffected
    Set dbs = Nothing
End Sub
