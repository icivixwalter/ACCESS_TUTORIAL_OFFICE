Esempio di proprietà DataUpdatable (MDB)

Nell'esempio riportato di seguito si creano due oggetti Recordset dalla stessa tabella, un oggetto Recordset di tipo dynaset e un oggetto Recordset di tipo snapshot. La routine verifica quindi la proprietà DataUpdatable per il campo Cognome in ogni oggetto Recordset. L'impostazione della proprietà DataUpdatable è True (-1) per l'oggetto Recordset di tipo dynaset e False (0) per l'oggetto Recordset di tipo snapshot.

Sub VerificaDatiAggiornabili()
    Dim dbs As Database
    Dim rstDynaset As Recordset, rstSnapshot As Recordset
    Dim fldDynaset As Field, fldSnapshot As Field

    ' Restituisce il riferimento al database corrente.
    Set dbs = CurrentDb
    ' Apre oggetto Recordset di tipo dynaset.
    Set rstDynaset = dbs.OpenRecordset("Impiegati", _
        dbOpenDynaset)
    ' Apre oggetto Recordset di tipo snapshot.
    Set rstSnapshot = dbs.OpenRecordset("Impiegati",_
        dbOpenSnapshot)
    ' Ottiene le variabili di oggetto Field che puntano a un campo di ogni oggetto Recordset.
    Set fldDynaset = rstDynaset.Fields!Cognome
    Set fldSnapshot = rstSnapshot.Fields!Cognome
    ' Ottiene il record corrente.
    rstDynaset.MoveFirst
    rstSnapshot.MoveFirst
    ' Visualizza il valore della proprietà DataUpdatable per ogni oggetto Recordset.
    Debug.Print "DatiAggiornabili (Recordset di tipo dynaset): "; _
        fldDynaset.DataUpdatable
    Debug.Print "DatiAggiornabili (Recordset di tipo snapshot): "; _
        fldSnapshot.DataUpdatable
    rstDynaset.Close
    rstSnapshot.Close
    Set dbs = Nothing
End Sub
