Esempio di proprietà RecordsAffected (MDB)

Nell'esempio riportato di seguito si stampa il numero di record aggiornati mediante una query di comando.

Sub AggiornamentoRecord()
    Dim dbs As Database, qdf As QueryDef
    Dim strSQL As String

    ' Restituisce il riferimento al database corrente.
    Set dbs = CurrentDb
    strSQL = "UPDATE Impiegati SET Posizione = " _
        & "'Caporappresentante' " _
        & "WHERE Posizione = 'Rappresentante';"
    ' Crea nuovo aggetto QueryDef.
    Set qdf = dbs.CreateQueryDef("AggiornaPosizioni", strSQL)
    ' Esegue oggetto QueryDef.
    qdf.Execute
    Debug.Print qdf.RecordsAffected
    Set dbs = Nothing
End Sub
