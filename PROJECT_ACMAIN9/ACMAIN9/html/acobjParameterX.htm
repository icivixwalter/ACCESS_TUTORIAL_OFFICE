Esempio di oggetto Parameter e insieme Parameters (MDB)

Nell'esempio riportato di seguito viene creata una nuova query con parametri e vengono forniti i valori per tali parametri:

Sub NuovaQueryConParametri()
    Dim dbs As Database, qdf As QueryDef, rst As Recordset
    Dim prm As Parameter, strSQL As String

    ' Restituisce il riferimento al database corrente.
    Set dbs = CurrentDb
    ' Costruisce la stringa SQL.
    strSQL = "PARAMETERS [DataOrdine iniziale] DataOra, " _
        & "[DataOrdine finale] DataOra; SELECT * FROM Ordini " & _
        "WHERE ([DataOrdine] Between[DataOrdine iniziale] " _
        & "And [DataOrdine finale]);"
    ' Crea il nuovo oggetto QueryDef.
    Set qdf = dbs.CreateQueryDef("QueryConParametri", strSQL)
    ' Fornisce i valori per i parametri.
    qdf.Parameters![DataOrdine iniziale] = #1/4/95#
    qdf.Parameters![DataOrdine finale] = #30/4/95#
    ' Apre il recordset su oggetto QueryDef.
    Set rst = qdf.OpenRecordset
    rst.MoveLast
    MsgBox "Query restituita " & rst.RecordCount & " record."
    rst.Close
    Set dbs = Nothing
End Sub
