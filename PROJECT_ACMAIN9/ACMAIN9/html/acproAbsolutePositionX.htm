Esempio di proprietà AbsolutePosition (MDB)

Nell'esempio riportato di seguito viene ricercato un determinato record in un oggetto Recordset di tipo dynaset e viene determinata la posizione di tale record.

Sub PosizioneRecord()
    Dim dbs As Database, rst As Recordset

    ' Restituisce il riferimento al database corrente.
    Set dbs = CurrentDb
    ' Restituisce il riferimento dell'oggetto Recordset di tipo dynaset.
    Set rst = dbs.OpenRecordset("SELECT * FROM Ordini WHERE " _
        & "DataSpedizione >= #1-1-96#;")
    ' Ricerca il primo ordine inviato a Londra.
    rst.FindFirst "Città = 'Londra'"
    ' Restituisce la posizione di questo record.
    Debug.Print rst.AbsolutePosition
    rst.Close
    Set dbs = Nothing
End Sub
