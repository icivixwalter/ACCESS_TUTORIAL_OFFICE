Esempio di proprietà Object

Nell'esempio seguente viene utilizzata la subroutine PrintObjectProperties per visualizzare i valori delle proprietà Object per un oggetto nella finestra Debug. La subroutine richiede come argomenti il tipo oggetto ed il nome oggetto.

Dim strTipoOggetto As String
Dim strNomeOggetto As String
Dim strMsg As String

strMsg = "Immettere Tipo oggetto (ad es. Forms, Scripts, " _
  & "Modules, Reports, Tables)."
' Ottiene tipo oggetto.
strTipoOggetto = InputBox(strMsg)
strMsg = "Immettere il nome di una maschera, macro, modulo, " _
  & "query, report, o finestra."
' Ottiene nome oggetto da utente.
StrNomeOggetto = InputBox(strMsg)
' Passa tipo oggetto e nome oggetto a
' subroutine PrintObjectProperties.
PrintObjectProperties strTipoOggetto, strNomeOggetto

Sub PrintObjectProperties(strTipoOggetto As String, strNomeOggetto _
  As String)
Dim dbs As Database, ctr As Container, doc As Document
Dim intI As Integer
Dim strCarTab As String
Dim prp As Property

Set dbs = CurrentDb
strCarTab = vbTab
' Imposta variabile oggetto Container.
Set ctr = dbs.Containers(strTipoOggetto)
' Imposta variabile oggetto Document.
Set doc = ctr.Documents(strNomeOggetto)
doc.Properties.Refresh
' Visualizza nome oggetto nella finestra Debug.
Debug.Print doc.Name
' Visualizza ogni proprietà di oggetto nella finestra Debug.
For Each prp in doc.Properties
    Debug.Print strCarTab & prp.Name & " = " & prp.Value
Next
End Sub
