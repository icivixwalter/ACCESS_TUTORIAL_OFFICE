Esempio di metodo InsertText

Nell'esempio riportato di seguito viene indicato come inserire una stringa di testo in un modulo standard:

Function InsertProc(strModuleName) As Boolean
    Dim mdl As Module, strText As String

    On Error GoTo Error_InsertProc
    ' Apre il modulo.
    DoCmd.OpenModule strModuleName
    ' Restituisce il riferimento a oggetto Module.
    Set mdl = Modules(strModuleName)
    ' Inizializza la variabile di stringa.
    strText = "Sub VisualizzaMessaggio()" & vbCrLf _
        & vbTab & "MsgBox ""Bello!""" & vbCrLf _
        & "End Sub"
    ' Inserisce testo nel modulo.
    mdl.InsertText strText
    InsertProc = True

Exit_InsertProc:
    Exit Function

Error_InsertProc:
    MsgBox Err & ": " & Err.Description
    InsertProc = False
    Resume Exit_InsertProc
End Function
