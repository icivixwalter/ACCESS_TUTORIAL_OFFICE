Esempio di metodo Print

Nell'esempio riportato di seguito viene indicato come utilizzare il metodo Print per visualizzare il testo su un report chiamato Report1. Per centrare il testo in verticale e orizzontale vengono utilizzati i metodi TextWidth e TextHeight.

Private Sub Formato_Dettagli(Cancel As Integer, _
        FormatCount As Integer)
    Dim rpt as Report
    Dim strMessage As String
    Dim intHorSize As Integer, intVerSize As Integer

    Set rpt = Me
    strMessage = "VisualizzaMessaggio"
    With rpt
        'Imposta la scala in pixel e le proprietà
        'FontName e FontSize.
        .ScaleMode = 3
        .FontName = "Courier"
        .FontSize = 24
    End With
    ' Larghezza in orizzontale.
    intHorSize = Rpt.TextWidth(strMessage)
    ' Altezza in verticale.
    intVerSize = Rpt.TextHeight(strMessage)
    ' Calcola la posizione del testo da visualizzare.
    Rpt.CurrentX = (Rpt.ScaleWidth/2) - (intHorSize/2)
    Rpt.CurrentY = (Rpt.ScaleHeight/2) - (intVerSize/2)
    ' Stampa il testo in oggetto Report.
    Rpt.Print strMessage
End Sub
