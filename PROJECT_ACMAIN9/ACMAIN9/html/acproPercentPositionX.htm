Esempio di proprietà PercentPosition (MDB)

Nell'esempio riportato di seguito viene indicata la posizione corrente in percentuale di un record in un set di record, quindi si visualizza tale posizione in una casella di messaggio.

Sub PercentualeDiRecord()
    Dim dbs As Database, rst As Recordset
    Dim strInput As String, strSQL As String

    strSQL = "SELECT * FROM Prodotti ORDER BY NomeProdotto;"
    ' Restituisce il riferimento al database corrente.
    Set dbs = CurrentDb
    ' Apre oggetto Recordset di tipo dynaset.
    Set rst = dbs.OpenRecordset(strSQL, dbOpenDynaset)
    ' Si sposta alla fine del set di record.
    rst.MoveLast
    ' Torna all'inizio del set di record.
    rst.MoveFirst
    ' Richiede di immettere il nome del prodotto.
    strInput = InputBox("Immettere il nome completo del prodotto.")
    ' Ricerca la prima ricorrenza.
    rst.FindFirst "NomeProdotto = '" & strInput & "'"
    ' Visualizza la posizione corrente in percentuale.
    MsgBox("La posizione corrente del set di record è " _
        & rst.PercentPosition & "%.")
    rst.Close
    Set dbs = Nothing
End Sub
