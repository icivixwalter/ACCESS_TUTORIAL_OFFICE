Esempio di oggetto Index e insieme Indexes (MDB)

Nell'esempio riportato di seguito viene creato un nuovo indice in una tabella Impiegati.

Sub NuovoIndice()
    Dim dbs As Database, tdf As TableDef, idx As Index
    Dim fld1 As Field, fld2 As Field

    ' Restituisce riferimento a database corrente.
    Set dbs = CurrentDb
    ' Restituisce riferimento a tabella Impiegati.
    Set tdf = dbs.TableDefs!Impiegati
    Set idx = tdf.CreateIndex("IndiceNomiImpiegati")
    Set fld1 = idx.CreateField("Cognome")
    Set fld2 = idx.CreateField("Nome")
    idx.Fields.Append fld1
    idx.Fields.Append fld2
    idx.Required = True
    tdf.Indexes.Append idx
    Set dbs = Nothing
End Sub
