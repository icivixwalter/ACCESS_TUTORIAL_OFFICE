Esempio di metodo CreateField (MDB)

Nell'esempio riportato di seguito si crea una nuova tabella con due nuovi campi. Uno di questi è un campo Contatore. La routine inoltre rende questo campo la chiave primaria della tabella.

Sub NuovaTabella()
    Dim dbs As Database
    Dim tdf As TableDef, fld1 As Field, fld2 As Field
    Dim idx As Index, fldIndex As Field

    ' Restituisce il riferimento al database corrente.
    Set dbs = CurrentDb
    ' Crea la nuova tabella con due campi.
    Set tdf = dbs.CreateTableDef("Contatti")
    Set fld1 = tdf.CreateField("IDContatto", dbLong)
    fld1.Attributes = fld1.Attributes + dbAutoIncrField
    Set fld2 = tdf.CreateField("NomeContatto", dbText, 50)
    ' Accoda i campi.
    tdf.Fields.Append fld1
    tdf.Fields.Append fld2
    ' Crea indice di chiave primaria.
    Set idx = tdf.CreateIndex("ChiavePrimaria")
    Set fldIndex = idx.CreateField("IDContatto", dbLong)
    ' Accoda i campi indice.
    idx.Fields.Append fldIndex
    ' Imposta la proprietà Primary.
    idx.Primary = True
    ' Accoda indice.
    tdf.Indexes.Append idx
    ' Accoda oggetto TableDef.
    dbs.TableDefs.Append tdf
    dbs.TableDefs.Refresh
    Set dbs = Nothing
End Sub
