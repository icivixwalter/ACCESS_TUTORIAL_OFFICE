Esempi di proprietà InsideHeight, InsideWidth

Nell'esempio riportato di seguito si indica come utilizzare le proprietà InsideHeight e InsideWidth per confrontare l'altezza e la larghezza interna di una maschera con l'altezza e la larghezza della finestra che la contiene. Se le dimensioni di quest'ultima non sono uguali a quelle della maschera, la finestra viene ridimensionata in modo che l'altezza e la larghezza corrispondano a quelle della maschera.

Sub ReimpostaDimFinestra(frm As Form)
    Dim intAltezzaFinestra As Integer
    Dim intLarghezzaFinestra As Integer
    Dim intAltezzaTotMaschera As Integer
    Dim intLarghezzaTotMaschera As Integer
    Dim intIntestAltezza As Integer
    Dim intDettaglioAltezza As Integer
    Dim intPièDiPAltezza As Integer

    ' Determina l'altezza della maschera.
    intIntestAltezza = frm.Section(acHeader).Height
    intDettaglioAltezza = frm.Section(acDetail).Height
    intPièDiPAltezza = frm.Section(acFooter).Height
    intAltezzaTotMaschera = intIntestazAltezza _
        + intDettaglioAltezza + intPièDiPAltezza
    ' Determina la larghezza della maschera.
    intLarghezzaTotMaschera = frm.Width
    ' Determina l'altezza e la larghezza della maschera.
    intAltezzaFinestra = frm.InsideHeight
    intLarghezzaFinestra = frm.InsideWidth

    If intLarghezzaFinestra <> intLarghezzaTotMaschera Then
        frm.InsideWidth = intLarghezzaTotMaschera
    End If
    If intAltezzaFinestra <> intAltezzaTotMaschera Then
        frm.InsideHeight = intAltezzaTotMaschera
    End If
End Sub
