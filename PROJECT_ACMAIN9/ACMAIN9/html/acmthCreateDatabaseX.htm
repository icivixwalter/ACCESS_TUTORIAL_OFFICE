Esempio di metodo CreateDatabase (MDB)

Nell'esempio riportato di seguito viene utilizzato il metodo CreateDatabase per creare un nuovo database crittografato chiamato Nuovodb.mdb:

Sub NuovoDatabase()
    Dim wspDefault As Workspace, dbs As Database
    Dim tdf As TableDef, fld1 As Field, fld2 As Field
    Dim idx As Index, fldIndex As Field

    Set wspDefault = DBEngine.Workspaces(0)
    ' Crea il nuovo database crittografato.
    Set dbs = wspDefault.CreateDatabase ("Nuovodb.mdb", _
        dbLangGeneral, dbEncrypt)
    ' Crea una nuova tabella con due campi.
    Set tdf = dbs.CreateTableDef("Contatti")
    Set fld1 = tdf.CreateField("IDContatto", dbLong)
    fld1.Attributes = fld1.Attributes + dbAutoIncrField
    Set fld2 = tdf.CreateField("Contatto", dbText, 50)
    ' Accoda i campi.
    tdf.Fields.Append fld1
    tdf.Fields.Append fld2
    ' Crea indice chiave primaria.
    Set idx = tdf.CreateIndex("ChiavePrimaria")
    Set fldIndice = idx.CreateField("IDContatto", dbLong)
    ' Accoda i campi indice.
    idx.Fields.Append fldIndex
    ' Imposta la proprietà Primary.
    idx.Primary = True
    ' Accoda indice.
    tdf.Indexes.Append idx
    ' Accoda oggetto TableDef.
    dbs.TableDefs.Append tdf
    dbs.TableDefs.Refresh
    Set dbs = Nothing
End Sub
