Esempio di metodo Delete

Nell'esempio riportato di seguito viene creata una colonna in una tabella, quindi tale colonna viene eliminata.

Sub EliminaColonna()
    Dim cat As New ADOX.Catalog, tbl As ADOX.Table
    
    ' Restituisce il riferimento al database corrente.
    Set cat.ActiveConnection = CurrentProject.Connection
    Set tbl = cat.Tables!Dipendenti
    ' Accoda il nuovo oggetto Column.
    tbl.Columns.Append "Patronimico", adVarWChar, 2
    tbl.Columns.Refresh
    ' Elimina il nuovo oggetto Column.
    tbl.Columns.Delete "Patronimico"
tbl.Columns.Refresh
    Set cat = Nothing
End Sub
