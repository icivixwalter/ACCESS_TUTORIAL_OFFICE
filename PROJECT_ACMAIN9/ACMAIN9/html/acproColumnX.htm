Esempio di proprietà Column e ColumnCount

Negli esempi riportati di seguito viene indicato come utilizzare le proprietà Column e ColumnCount per stampare i valori selezionati in una casella di riepilogo.

Sub Leggi_ListBox()
    Dim intNumColonne As Integer, inti As Integer
    Dim frmClienti As Form

    Set frmClienti = Forms!frmClienti
    If frmClienti!lstNomiClienti.ItemsSelected.Count > 0 Then
        ' Qualsiasi selezione?
        intNumColonne = frmClienti!lstNomiClienti.ColumnCount
        Debug.Print "La casella di riepilogo contiene"; intNumColonne; _
            IIf(intNumColonne = 1, " colonna", " colonne"); _
             " di dati."
        Debug.Print "La selezione corrente contiene:"
        For inti = 0 To intNumColonne - 1
            ' Print column data.
            Debug.Print frmClienti!lstNomiClienti.Column(inti)
        Next inti
    Else
        Debug.Print "Non è stato selezionato un elemento " _
            &"della casella di riepilogo."
    End If
End Sub
