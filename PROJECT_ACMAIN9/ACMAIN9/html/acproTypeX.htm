Esempio di proprietà Type

Nell'esempio riportato di seguito si determina se un oggetto Module rappresenti un modulo standard o un modulo di classe.

Function VerificaTipoModulo(strNomeModulo As String) As Integer
    Dim mdl As Module

    ' Apre il modulo da includere nell'insieme Modules.
    DoCmd.OpenModule strNomeModulo
    ' Restituisce il riferimento all'oggetto Module.
    Set mdl = Modules(strNomeModulo)
    ' Controlla la proprietà Type.
    If mdl.Type = acClassModule Then
        ' Inserisce il commento.
        mdl.InsertLines 1, "' Modulo di classe."
        VerificaTipoModulo = acClassModule
    Else
        ' Inserisce il commento.
        mdl.InsertLines 1, "' Modulo standard."
        VerificaTipoModulo = acModuloStandard
    End If
End Function
