Esempi di proprietà Error, PageCount, SnapshotPath e di evento ReadyStateChange

Nell'esempio seguente viene utilizzata un'istruzione Select Case in una routine evento ReadyStateChange di controllo Snapshot per determinare lo stato della proprietà ReadyState. Se il caricamento di un file snapshot nel controllo ha esito positivo, viene visualizzata una finestra di messaggio che riporta le informazioni sul percorso del file e sul conteggio delle relative pagine. Se viceversa durante il caricamento dello snapshot si verifica un errore, viene chiamata una routine personalizzata di gestione degli errori per visualizzare le informazioni sull'errore che si è verificato. All'inizio della procedura, vengono definite le costanti per ogni possibile valore della proprietà ReadyState.

Private Sub Snapshot1_ReadyStateChange(ByVal ReadyState As Long)
  Const conSnpNotInitialized = 0
  Const conSnpLoading = 1
  Const conSnpStillDownloading = 2
  Const conSnpDownloadComplete = 4
  Select Case ReadyState
    Case conSnpNotInitialized
      ' controllo snapshot non inizializzato.
    Case conSnpLoading
      ' Controllo snapshot in corso di caricamento.
    Case conSnpStillDownloading
      ' Controllo snapshot caricato. File
      ' snapshot in corso di caricamento.
    Case conSnpDownloadComplete
      ' Caricamento snapshot completato.
    With Snapshot1
      If.Error = 0 Then
        ' File snapshot caricato nel controllo.
          Msgbox.SnapshotPath & _
          " caricato nel " _
          & "controllo Snapshot Viewer." & vbCrLf & vbCrLf _
          & "Il file snapshot contiene " &.PageCount _
          & " pagine."
      Else
        ' Si è verificato un errore durante il caricamento del file snapshot.
        Call HandleSnapError(.SnapshotPath,.Error)
        Exit sub
      End If
    End With
  End Select
End Sub

Sub HandleSnapError(strFile As String, lngError As Long)
  Dim strError As String
  Select Case lngError
    Case 2500
      strError = "Errore " & CStr(lngError) & ": Valore non valido."
    Case 2501
      strError = "Errore " & CStr(lngError) & ": Pagina non valida."
    Case 2502
      strError = "Errore " & CStr(lngError) & _
        ": File snapshot non valido o mancante."
    Case 2503
      strError = "Errore " & CStr(lngError) & _
        ": Proprietà di sola lettura."
    Case 2504
      strError = "Errore " & CStr(lngError) & _
        ": Operazione annullata dall'utente."
    Case 2505
      strError = "Errore " & CStr(lngError) & _
        ": Per visualizzare questo file snapshot è necessaria " _
        & "una versione più recente del controllo Snapshot Viewer."
    Case 2506
      strError = "Errore " & CStr(lngError) & _
        ": Errore durante la stampa."
    Case 2507
      strError = "Errore " & CStr(lngError) & _
        ": Errore durante l'apertura di un file snapshot."
    Case 2508
      strError = "Errore " & CStr(lngError) & ": Memoria esaurita"
    Case 2509
      strError = "Errore " & CStr(lngError) & _
        ": Si è verificato un errore di scrittura in un " _
        & "file temporaneo durante l'apertura di un file snapshot."
    Case 2510
      strError = "Errore " & CStr(lngError) & _
        ": Non vi è una stampante installata oppure " _
        & "nessuna stampante è stata impostata come predefinita."
    Case Else
      strError = "Errore " & CStr(lngError) & _
        ": Si è verificato un errore sconosciuto."
  End Select

  MsgBox strError & " File: " & strFile
End Sub
