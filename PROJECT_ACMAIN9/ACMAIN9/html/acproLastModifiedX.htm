Esempio di proprietà LastModified (MDB)

Nell'esempio riportato di seguito si determina l'ultimo record modificato in un set di record e si imposta un segnalibro per tale record.

Sub UltimoRecordModificato()
    Dim dbs As Database, rst As Recordset, varLibro As Variant

    ' Restituisce il riferimento al database corrente.
    Set dbs = CurrentDb
    ' Crea oggetto Recordset di tipo dynaset.
    Set rst = dbs.OpenRecordset("SELECT * " _
        & "FROM Ordini ORDER BY PaeseDestinatario;")
    With rst
       .FindFirst "PaeseDestinatario = 'UK'"
        ' Modifica il record e salva le modifiche.
       .Edit
        !PaeseDestinatario = "USA"
       .Update
    End With
    ' Verifica se il set di record supporta i segnalibri.
    If rst.Bookmarkable Then
        ' Imposta il segnalibro all'ultimo record modificato.
        varLibro = rst.LastModified
        ' Imposta la valuta all'ultimo record modificato.
        rst.Bookmarks = varLibro
    End If
    rst.Close
    Set dbs = Nothing
End Sub
