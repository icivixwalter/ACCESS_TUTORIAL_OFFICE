Esempio di proprietà NoMatch (MDB)

Nell'esempio riportato di seguito si utilizza la proprietà NoMatch per determinare se un processo del metodo FindFirst è stato eseguito in modo corretto.

Function TrovaPaese() As Integer
    Dim dbs As Database, rst As Recordset
    Dim strPaese As String

    ' Restituisce il riferimento al database corrente.
    Set dbs = CurrentDb
    ' Crea un oggetto Recordset di tipo dynaset.
    Set rst = dbs.OpenRecordset("Ordini", dbOpenDynaset)
    strPaese = InputBox("Immettere il nome di un Paese.")
    rst.FindFirst "PaeseDestinatario = '" & strPaese & "'"
    If rst.NoMatch Then
        TrovaPaese = False
    Else
        TrovaPaese = True
        Debug.Print rst!IDOrdine
    End If
    rst.Close
    Set dbs = Nothing
End Function
