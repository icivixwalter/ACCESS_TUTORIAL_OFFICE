Esempio di file modello HTML

<HTML>

<!--Il token seguente colloca il nome dell'oggetto nella barra del titolo del browser Web.-->

<TITLE><!--AccessTemplate_Titolo--></TITLE>

<!--Il tag HTML seguente crea un colore di sfondo diverso da quello predefinito del browser Web.-->

<BODY BACKGROUND = "grigio.jpg">

<!--Il token seguente colloca tutto l'output degli oggetti all'interno del tag <BODY>.-->

<!--AccessTemplate_Corpo-->

</BODY>

<BR><BR>

<!--I quattro token seguenti creano quattro pulsanti di testo che consentono di passare alla prima e pagina di un report, alla pagina precedente, a quella successiva o all'ultima.-->

<A HREF = "<!--AccessTemplate_PrimaPagina-->">Prima</A>

<A HREF = "<!--AccessTemplate_PaginaPrecedente-->">Precedente</A>

<A HREF = "<!--AccessTemplate_PaginaSuccessiva-->">Successiva</A>

<A HREF = "<!--AccessTemplate_UltimaPagina-->">Ultima</A>

<!--Il token seguente inserisce il testo "Pagina n", dove n è il numero di pagina corrente.-->

<P ALIGN = CENTER>Page <!--AccessTemplate_NumeroPagina-->.</P>

<!--Il tag HTML seguente aggiunge il logo della società alla fine della pagina Web.-->

<IMG SRC = "logo_società.jpg">

</HTML>
