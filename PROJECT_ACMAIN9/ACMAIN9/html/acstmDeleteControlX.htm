Esempio di metodi DeleteControl e DeleteReportControl

Nell'esempio riportato di seguito si crea una maschera con un pulsante di comando e si visualizza un messaggio che chiede all'utente se desidera eliminare il pulsante di comando. Se si sceglie Sì, il pulsante di comando viene eliminato.

Sub EliminaPulsanteComando()
    Dim frm As Form, ctlNuovo As Control
    Dim strMsg As String, intRisposta As Integer, intDialogo As Integer

    ' Crea una nuova maschera e punta a essa.
    Set frm = CreateForm
    ' Crea un nuovo pulsante di comando.
    Set ctlNuovo = CreateControl(frm.Name, acCommandButton)
    ' Ripristina la maschera.
    DoCmd.Restore
    ' Imposta etichetta.
    ctlNuovo.Caption = "Nuovo pulsante di comando"
    ' Ridimensiona il controllo.
    ctlNuovo.SizeToFit
    ' Richiede all'utente di eliminare il controllo.
    strMsg = "Eliminazione di " & ctlNuovo.Name &". Continua?"
    ' Definisce i pulsanti da visualizzare nella finestra di dialogo.
    intDialogo = vbYesNo + vbCritical + vbDefaultButton2
    intRisposta = MsgBox(strMsg, intDialogo)
    If intRisposta = vbYes Then
        ' Elimina il controllo.
        DeleteControl frm.Name, ctlNuovo.Name
    End If
End Sub
