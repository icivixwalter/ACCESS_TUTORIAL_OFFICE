Esempio di proprietà ControlType, AllowAdditions, AllowDeletions, AllowEdits

Nell'esempio riportato di seguito si esamina la proprietà ControlType di tutti i controlli di una maschera. Per ciascun controllo casella di testo ed etichetta, la routine alterna la proprietà SpecialEffect per tali controlli. Quando la proprietà SpecialEffect dei controlli etichetta è impostata a Ombreggiato, la proprietà SpecialEffect dei controlli casella di testo è impostata a Normale e le proprietà AllowAdditions, AllowDeletions e AllowEdits sono tutte impostate a True, la variabile intConsenteModifica viene alternata per consentire la modifica dei dati sottostanti.

Sub Interruttore(frm As Form)
    Dim ctl As Control
    Dim intI As Integer, intConsenteModifica As Integer
    Const conTrasparente = 0
    Const conBianco = 16777215
    For Each ctl in frm.Controls
        With ctl
            Select Case.ControlType
                Case acLabel
                    If.SpecialEffect = acEffectShadow Then
                       .SpecialEffect = acEffectNormal
                       .BorderStyle = conTrasparente
                        intConsenteModifica = True
                    Else
                       .SpecialEffect = acEffectShadow
                        intConsenteModifica = False
                    End If
                Case acTextBox
                    If.SpecialEffect = acEffectNormal Then
                       .SpecialEffect = acEffectSunken
                       .BackColor = conBianco
                    Else
                       .SpecialEffect = acEffectNormal
                       .BackColor = frm.Corpo.BackColor
                    End If
            End Select
        End With
    Next ctl
    If intConsenteModifica = False Then
        With frm
           .AllowAdditions = False
           .AllowDeletions = False
           .AllowEdits = False
        End With
    Else
        With frm
           .AllowAdditions = True
           .AllowDeletions = True
           .AllowEdits = True
        End With
    End If
End Sub
